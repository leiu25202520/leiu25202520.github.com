date:   2016-04-10 07:56:25 +0800
---


{% highlight ruby %}
#不冲突
vim .git/config
vim .git/refs/remote/origin/master
git fetch origin
git rebase origin/master
git push
{% endhighlight %}


{% highlight ruby %}
＃冲突文件 先拉下来，然后 修改冲突
git fetch origin
git add .
git rebase --continue
git push
{% endhighlight %}
